git - система контроля
github  - сервис платформа спйт предоставляющий хранилище для версий


sudo apt git - установка системы контроля версий

git --version - проверка версии

git config --global user.email "your @gmail.com"
git config --global user.name "yourName"

git config --list - проверка настроек

ssh-keygen - создали ssh ключ

cat ~/.ssh/id_rsa.pub - получение ssh ключа

создвем репозиторий на git hub

переходим в папку с проектом
git init - превращает в текущую диреткорию в git репозиторий
git remote add название_свзяи(обычно - origin) сылку_на-удаленный_репозиторий(SSH: git@github:...)
git remote -v - просмотр списка удаленных репозиторов
git remote -rm название_связи
git status - проверка статуса файлов
git add название_файлов - отслеживание конкретных файлов
git add . - отслеживание всех файлов


git commit - создание коммита(версии/слепка кода)
git commit -m описание версии - создание комитта с коментарием
git push название_связи(обычно-origin) отправка версии кода на удаленный репозиторий
Например: git push origin master


git clone сылка_на_удаленный_репозитор - склонировать/мкопировать/скачать удаленный репозиторий

git diff - просмотрт внесенных изменений 

git branch - вывод списка веток в терминал
(звездочкой помечается активная ветка)

git branch название_ветки - на указанной ветке

git checkout название_ветки
                           - переход на указанную ветку
git switch название ветки


git pull название_связи название_ветки - стягивание изменений с удаленного репозитория, с указанной ветки


git stash - спрятать локальные данные во временную область
git stash list - просмотр списка спрятанных данных
git stash apply - запись спрятвнных данных/извлечение спрятанных данных
git stash - удаление спрятанных данных

git log - просмотр коммитов/версий
git log --one line - компактое отображение коммитов
